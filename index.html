function getDiceRoll(serverSeed, clientSeed, nonce) {
  var hash = sha512(serverSeed + clientSeed + nonce);
  var index = 0;
  do {
    var lucky = parseInt(hash.substr(index, 5), 16);
    index += 5;
  } while (lucky >= 1000000);
  return lucky % 10000;
}
